import time
def dp(filename, budget_million):
    df = pd.read_excel(filename)
    names = df['‡πÇ‡∏Ñ‡∏£‡∏á‡∏Å‡∏≤‡∏£'].tolist()
    costs = df['‡∏á‡∏ö‡∏õ‡∏£‡∏∞‡∏°‡∏≤‡∏ì (‡∏•‡πâ‡∏≤‡∏ô‡∏ö‡∏≤‡∏ó)'].tolist()
    profits = df['‡∏ú‡∏•‡∏ï‡∏≠‡∏ö‡πÅ‡∏ó‡∏ô (%)'].tolist()
    n = len(costs)
    W = int(budget_million)
   dp = [[0] * (W + 1) for _ in range(n + 1)]
   start_time = time.time() # ‡πÄ‡∏£‡∏¥‡πà‡∏°‡∏à‡∏±‡∏ö‡πÄ‡∏ß‡∏•‡∏≤
   for i in range(1, n + 1):
       for w in range(W + 1):
          if costs[i - 1] <= w:
             dp[i][w] = max(dp[i - 1][w],
             dp[i - 1][int(w - costs[i - 1])] + profits[i - 1])
          else:
             dp[i][w] = dp[i - 1][w]
    w = W
    selected = []
    for i in range(n, 0, -1):
       if dp[i][w] != dp[i - 1][w]:
          selected.append(i - 1)
          w -= int(costs[i - 1])
    end_time = time.time() # ‡∏à‡∏ö‡∏à‡∏±‡∏ö‡πÄ‡∏ß‡∏•‡∏≤
    elapsed_time = end_time - start_time
    selected.reverse()

    print("‡∏ú‡∏•‡∏ï‡∏≠‡∏ö‡πÅ‡∏ó‡∏ô‡∏™‡∏π‡∏á‡∏™‡∏∏‡∏î:", dp[n][W], "%")
